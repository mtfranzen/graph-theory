		\begin{theorem}{A bipartite complete graph $K_{m, n}$ is Hamiltonian iff $m = n$. It's girth is $4$ for $m, n \geq 2$ and $\infty$ otherwise. It's diameter is $2$. The graph's order is $m + n$ and it's size is $m \cdot n$.}
		\end{theorem}
